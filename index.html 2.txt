<script>
  function showSlide(num) {
    for (let i = 1; i <= 6; i++) {
      const s = document.getElementById("slide" + i);
      if (s) s.classList.add("hidden");
    }
    document.getElementById("slide" + num).classList.remove("hidden");
  }

  let count = 0;
  let interval;

  function startGame() {
    count = 0;
    document.getElementById("counter").innerText = "0 / 10";
    showSlide(4);
    const area = document.getElementById("gameArea");
    area.innerHTML = "";

    function spawnHeart() {
      const heart = document.createElement("div");
      heart.innerHTML = "ğŸ’—";
      heart.className = "heart";
      heart.style.left = Math.random() * 85 + "%";
      heart.style.top = Math.random() * 75 + "%";

      heart.onclick = () => {
        heart.remove();
        count++;
        document.getElementById("counter").innerText = count + " / 10";
        if (count >= 10) {
          clearInterval(interval);
          showSlide(5);
        }
      };
      area.appendChild(heart);

      setTimeout(() => {
        if (heart.parentNode) heart.remove();
      }, 1200);
    }

    interval = setInterval(spawnHeart, 700);
  }
</script>
